grammar Lenguaje;

// --- REGLAS SINTÁCTICAS (Parser) ---
programa: funcion+ EOF;

funcion: 'funcion' ID '(' parametros? ')' '{' instrucciones* '}';

parametros: ID (',' ID)*;

// Como dice que instrucciones es una lista de leer, escribir o asignacion:
instrucciones: (leer | escribir | asignacion);

leer: 'leer' '(' ID ')' ';';

escribir: 'escribir' '(' expresion ')' ';';

asignacion: ID '=' expresion ';';

// Acá le agregamos la suma (+) para que el ejemplo del profe funcione
expresion: expresion '+' expresion
         | NUMERO
         | CADENA
         | ID
         ;

// --- REGLAS LÉXICAS (Lexer) ---
NUMERO: [0-9]+;
ID: [a-zA-Z_] [a-zA-Z0-9_]*;
CADENA: '"' ~["]* '"'; // Esto lee cualquier cosa entre comillas dobles

// Ignorar espacios en blanco y saltos de línea
WS: [ \t\r\n]+ -> skip;
